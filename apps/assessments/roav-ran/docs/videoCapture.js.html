var desktop = document.documentElement;

/**
 * Opens the browser in fullscreen mode.
 * @function
 */
export function openFullscreen() {
  if (desktop.requestFullscreen) {
    desktop.requestFullscreen();
  } else if (desktop.webkitRequestFullscreen) {
    /* Safari */
    desktop.webkitRequestFullscreen();
  } else if (desktop.msRequestFullscreen) {
    /* IE11 */
    desktop.msRequestFullscreen();
  }
}

/** @type {number} Countdown value */
export var countdown = 4;

/**
 * Updates the countdown and executes a callback when it reaches zero.
 * @function
 * @param {Function} callbackFunction - Function to call when countdown reaches zero.
 */
export function updateCountdown(callbackFunction) {
  countdown--;

  if (countdown === 0) {
    document.getElementById("instruction").style.display = "none";
    if (typeof callbackFunction === "function") {
      callbackFunction();
      countdown = 4;
    }
  } else {
    document.getElementById("instruction").innerHTML =
      "<h1>" + countdown + "</h1>";
    setTimeout(function () {
      updateCountdown(callbackFunction);
    }, 1000);
  }
}

/**
 * Requests access to the user's camera and optionally microphone.
 * @async
 * @function
 * @param {boolean} [audio=false] - Whether to request audio access.
 * @returns {Promise<void>}
 */
export async function giveAccess() {
  try {
    const stream = await navigator.mediaDevices.getUserMedia({
      video: {
        width: { ideal: 1920 },
        height: { ideal: 1080 },
        frameRate: { ideal: 60, max: 60 },
      },
      audio: true,
    });

    // Assign the camera stream to the inputVideo element
    if (inputVideo) {
      inputVideo.srcObject = stream;
      inputVideo.muted = true; // Mute the video element to prevent audio playback
      inputVideo.play();
      inputVideo.addEventListener("loadedmetadata", function () {
        orig_img_width = inputVideo.videoWidth;
        orig_img_height = inputVideo.videoHeight;
        if (headCanvas) {
          headCanvas.width = orig_img_width;
          headCanvas.height = orig_img_height;
        }
      });
    }

    // Create video+audio recorder
    videoRecorder = new MediaRecorder(stream, {
      mimeType: "video/webm",
    });
    videoRecorder.ondataavailable = (event) => {
      console.log("Video chunk size:", event.data.size);
      if (event.data.size > 0) {
        videoChunks.push(event.data);
      }
    };

    // If audio is enabled, create a separate audio-only recorder
    // Create a new MediaStream with only the audio track
    const audioStream = new MediaStream(stream.getAudioTracks());

    audioRecorder = new MediaRecorder(audioStream, {
      mimeType: "audio/webm",
    });
    audioRecorder.ondataavailable = (event) => {
      console.log("Audio chunk size:", event.data.size);

      if (event.data.size > 0) {
        audioChunks.push(event.data);
      }
    };

    // Modify the stop handler
    videoRecorder.onstop = async () => {
      console.log("Video recording stopped, blob created");
    };

    audioRecorder.onstop = async () => {
      console.log("Audio recording stopped, blob created");
    };
  } catch (error) {
    console.error("Error accessing media devices:", error);
  }
}

/**
 * Gets the upload URL for a file.
 * @async
 * @function
 * @param {string} fileName - Name of the file to upload.
 * @param {string} mimeType - MIME type of the file.
 * @param {string} parentDir - Parent directory for the file.
 * @returns {Promise<string>} The upload URL.
 */
export async function getUploadUrl(fileName, mimeType, parentDir) {
  try {
    const response = await fetch(
      `https://us-central1-webapp-af756.cloudfunctions.net/getUploadUrlFunction?fileName=${fileName}&mimeType=${mimeType}&parentDir=${parentDir}`,
    );
    // const response = await fetch(`http://127.0.0.1:5001/webapp-af756/us-central1/getUploadUrlFunction?fileName=${fileName}&mimeType=${mimeType}&parentDir=${parentDir}`);

    if (!response.ok) {
      throw new Error("Failed to fetch upload URL");
    }

    const result = await response.json();
    return result.result; // Assuming your Cloud Function response structure has a 'result' field
  } catch (error) {
    console.error("Error fetching upload URL:", error);
    throw error;
  }
}


export async function uploadFile(file, uploadUrl) {
  try {
    const response = await fetch(uploadUrl, {
      method: "PUT",
      headers: {
        "Content-Type": file.type,
        "Content-Length": file.size,
      },
      body: file,
    });

    if (!response.ok) {
      // Extract more details from the response
      const errorDetails = await response.text();
      throw new Error(
        `Upload failed: ${response.status} ${response.statusText}\n` +
          `Response headers: ${JSON.stringify(response.headers)}\n` +
          `Response body: ${errorDetails}`,
      );
    }

    console.log("Upload complete:", await response.text());
  } catch (error) {
    console.error("Error:", error);
  }
}
