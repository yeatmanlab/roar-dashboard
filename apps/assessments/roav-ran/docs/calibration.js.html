/**
 * Initializes the FaceMesh model with external resources.
 * @constant {FaceMesh}
 */
export const faceMesh = new FaceMesh({
  locateFile: (file) => {
    return `https://cdn.jsdelivr.net/npm/@mediapipe/face_mesh@0.4/${file}`;
  },
});

/**
 * Extracts eye coordinates from face mesh landmarks and updates global variables.
 * 
 * @param {Object} results - The results from the face mesh detection.
 * @param {Array<Object>} results.multiFaceLandmarks - An array where each entry contains the landmarks for a detected face.
 */
export function onResultsFaceMesh(results) {
  var leftEyeCoor = [];
  var rightEyeCoor = [];
  if (results.multiFaceLandmarks) {
    for (const landmarks of results.multiFaceLandmarks) {
      const indicesLeft = [130, 27, 243, 23];
      indicesLeft.forEach((index) => {
        const point = landmarks[index];
        leftEyeCoor.push([point.x, point.y]);
      });
      const indicesRight = [463, 257, 359, 253];
      indicesRight.forEach((index) => {
        const point = landmarks[index];
        rightEyeCoor.push([point.x, point.y]);
      });
    }
  }

  leftEyeCoordinates = leftEyeCoor;
  rightEyeCoordinates = rightEyeCoor;
}

// Configure FaceMesh options
faceMesh.setOptions({
  maxNumFaces: 1,
  refineLandmarks: true,
  minDetectionConfidence: 0.5,
  minTrackingConfidence: 0.5,
});

// Register the callback for FaceMesh processing
faceMesh.onResults(onResultsFaceMesh);

/**
 * Fetches an upload URL for saving files to cloud storage.
 * @async
 * @param {string} fileName - Name of the file to upload.
 * @param {string} mimeType - The file's MIME type.
 * @param {string} parentDir - The cloud directory.
 * @returns {Promise<string>} The upload URL.
 */
export async function getUploadUrl(fileName, mimeType, parentDir) {
  try {
    const response = await fetch(
      `https://us-central1-webapp-af756.cloudfunctions.net/getUploadUrlFunction?fileName=${fileName}&mimeType=${mimeType}&parentDir=${parentDir}`,
    );
    // const response = await fetch(`http://127.0.0.1:5001/webapp-af756/us-central1/getUploadUrlFunction?fileName=${fileName}&mimeType=${mimeType}&parentDir=${parentDir}`);

    if (!response.ok) {
      throw new Error("Failed to fetch upload URL");
    }

    const result = await response.json();
    return result.result; // Assuming your Cloud Function response structure has a 'result' field
  } catch (error) {
    console.error("Error fetching upload URL:", error);
    throw error;
  }
}

/**
 * Uploads a file to a cloud storage URL.
 * @async
 * @param {Blob} file - The file to upload.
 * @param {string} uploadUrl - The URL to upload to.
 */
export async function uploadFile(file, uploadUrl) {
  try {
    const response = await fetch(uploadUrl, {
      method: "PUT",
      headers: {
        "Content-Type": file.type,
        "Content-Length": file.size,
      },
      body: file,
    });

    if (!response.ok) {
      throw new Error("Upload failed: " + response.statusText);
    }

    console.log("Upload complete:", await response.text());
  } catch (error) {
    console.error("Error:", error);
  }
}

/**
 * Countdown timer used for UI updates.
 * @global
 * @type {number}
 */
export var countdown = 4;

/**
 * Updates the countdown display.
 * @param {Function} callbackFunction - Function to execute when countdown reaches zero.
 */
export function updateCountdown(callbackFunction) {
  countdown--;

  if (countdown === 0) {
    document.getElementById("instruction").style.display = "none";
    if (typeof callbackFunction === "function") {
      callbackFunction();
      countdown = 4;
    }
  } else {
    document.getElementById("instruction").innerHTML =
      "<h1>" + countdown + "</h1>";
    setTimeout(function () {
      updateCountdown(callbackFunction);
    }, 1000);
  }
}

/**
 * Preprocesses image data for neural network input.
 * 
 * Converts image data from RGBA format to normalized RGB format (0 to 1),
 * then rearranges the data to match the required model input shape.
 * 
 * @param {Uint8Array} data - The raw image pixel data in RGBA format.
 * @param {number} width - The width of the image.
 * @param {number} height - The height of the image.
 * @returns {Float32Array} The processed image data formatted for neural network input.
 */
export function preprocess(data, width, height) {
  const dataFromImage = ndarray(new Float32Array(data), [width, height, 4]);
  const dataProcessed = ndarray(new Float32Array(width * height * 3), [
    1,
    3,
    height,
    width,
  ]);

  // Normalize 0-255 to 0 - 1
  ops.divseq(dataFromImage, 255.0);
  // Realign imageData from [224*224*4] to the correct dimension [1*3*224*224].
  ops.assign(
    dataProcessed.pick(0, 0, null, null),
    dataFromImage.pick(null, null, 2),
  );
  ops.assign(
    dataProcessed.pick(0, 1, null, null),
    dataFromImage.pick(null, null, 1),
  );
  ops.assign(
    dataProcessed.pick(0, 2, null, null),
    dataFromImage.pick(null, null, 0),
  );
  return new Float32Array(dataProcessed.data);
}

/**
 * Preprocesses keypoint data for neural network input.
 * 
 * @param {Array<number>} data - The raw keypoint data.
 * @param {number} width - The width of the image.
 * @param {number} height - The height of the image.
 * @returns {Float32Array} The processed keypoint data.
 */
export function preprocess_kps(data, width, height) {
  const dataFromImage = ndarray(new Float32Array(data), [data.length]);
  const dataProcessed = ndarray(new Float32Array(data.length), [
    1,
    data.length,
  ]);
  ops.assign(dataProcessed.pick(0, null), dataFromImage);

  return new Float32Array(dataProcessed.data);
}

/**
 * Converts bounding box coordinates into YOLO format.
 * 
 * Transforms a feature's bounding box from absolute coordinates into a 
 * relative YOLO-compatible format, which normalizes values to a scale of 0-1.
 * 
 * @param {Array<Array<number>>} feature - The bounding box feature coordinates.
 * @param {number} w - The width of the image.
 * @param {number} h - The height of the image.
 * @returns {Array<number>} The converted bounding box in [x, y, width, height] format.
 */
export function convertToYolo(feature, w, h) {
  let centerx = (feature[0][0] + feature[2][0]) / 2.0;
  let centery = (feature[1][1] + feature[3][1]) / 2.0;
  let width = feature[2][0] - feature[0][0];
  let height = feature[3][1] - feature[1][1];
  let box = [centerx, centery, width, height];
  let upper_left_x = (box[0] - box[2] / 2) * w;
  let upper_left_y = (box[1] - box[3] / 2) * h;
  width = box[2] * w;
  height = box[3] * h;
  return [upper_left_x, upper_left_y, width, height];
}
